374 bool isColliding(int boxPosX, int boxPosY, int boxWidth, int boxHeight, Player *player)
375 {
376  if ((player->state.x + player->image.frameWidth > boxPosX) &&
377  (player->state.x < boxPosX + boxWidth) &&
378  (player->state.y + player->image.frameHeight > boxPosY) &&
379  (player->state.y < boxPosY + boxHeight - (player->image.frameHeight/2)))
380  {
381  if(player->state.idleE == true) player->state.x += 1;
382  if(player->state.idleD == true) player->state.x -= 1;
383  if(player->state.idleC == true) player->state.y += 1;
384  if(player->state.idleB == true) player->state.y -= 1;
385  return true;
386  }
387  else
388  {
389  return false;
390  }
